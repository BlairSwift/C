## 6. 计算几个数的和

```c
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#define number 10   //#define 定义的标示符常量
int main() 
{
	//C语言语法规定，变量要定义在当前代码块的最前面
	int num1;
	int num2;
    const int num3 = 1 ;//const-常量属性，即num3无法修改
	int sum;
	scanf("%d%d", &num1, &num2); //取地址符号&
	sum = num1 + num2 + num3 + number;
	printf("sum = %d\n", sum);
	return 0;
}
当有多个输入值时，空格和回车键都表示输入下一个
```



## 7. 移位操作符

```c
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
int main() 
{
	//移（2进制）位操作符：<<左移 >>右移
	int a = 1;
	// 整形1占4个字节 = 32bit位
	// 00000000000000000000000000000001
	int b = a << 2;
	// 00000000000000000000000000000100
	printf("%d\n", b); //4
    printf("%d\n", 1); //1
	return 0;
}
```

## 10. 运行监视

```c
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
void test()
{
	static int a = 1;
	a++;
	printf("a = %d\n", a);
}
int main() 
{
	int i = 0;
	while (i<5)
	{
		test();
		i++;
	}
	return 0;
}
按Ctrl+Fn+F10 , 再按调试，窗口，监视，再将监视的项加入名称，最后重复按Fn+F11.

运行结果：
a = 1
a = 2
a = 3
a = 4
a = 5

```

注意：1. static修饰局部变量：使局部变量的生命周期变长，static修饰全局变量：让静态的局部变量只能在自己的源文件内部中运行

2. static修饰函数：改变了函数的外部链接属性，函数只能在自己的源文件内部中运行

##  11. 指针

```c
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int main() 
{
	int a = 10;   //申请了4个字节的空间
	int* p = &a;  //p是一个指针变量
	printf("%p\n", p);//0000001D51D7F524
	printf("%d\n", sizeof(p));//指针在32位bit架构上都是占4字节，64位bit架构上占8字节
	*p = 20;      //*-解引用操作符/间接访问操作符
	printf("a = %d\n", a); //20
	printf("a = %d\n", *p);//20
	return 0;
}
```



##  12. 结构体

```c
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
//创建一个结构体类型
struct Book
{
	char name[20];
	short price;
};
//利用结构体类型创建一个该类型的结构体变量
int main() 
{
	struct Book b1 = {"C语言程序设计",50};
	//1. 利用结构体变量打印元素
	printf("书名：%s,价格：%d\n", b1.name,b1.price);
	b1.price = 30;
	printf("修改后的价格：%d\n",b1.price);
	//2. 利用指针打印元素
	struct Book* p = &b1;
	printf("书名：%s,价格：%d\n", p->name, p->price);
	return 0;
}
```
